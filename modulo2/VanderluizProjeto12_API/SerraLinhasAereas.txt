CREATE DATABASE SERRA_LINHAS_AEREAS
GO
USE SERRA_LINHAS_AEREAS
GO
CREATE TABLE CLIENTE (
    CPF BIGINT NOT NULL,
    NOME NVARCHAR(200) NOT NULL,
    SOBRENOME NVARCHAR(200) NOT NULL,
    RUA NVARCHAR(200) NOT NULL,
    NUMERO INT NOT NULL,
    BAIRRO NVARCHAR(100) NOT NULL,
    CEP NVARCHAR(100) NOT NULL,
    COMPLEMENTO NVARCHAR(250) NOT NULL,
    CONSTRAINT PK_CPF PRIMARY KEY (CPF)
)
GO
CREATE TABLE PASSAGEM(
    ID_PASSAGEM INT IDENTITY(1,1) NOT NULL,
    ORIGEM NVARCHAR(200) NOT NULL,
    DESTINO NVARCHAR(200) NOT NULL,
    DATA_ORIGEM DATE NOT NULL,
    DATA_DESTINO DATE NOT NULL,
    VALOR FLOAT NOT NULL
    CONSTRAINT PK_ID_PASSAGEM PRIMARY KEY (ID_PASSAGEM)
)
GO
CREATE TABLE VIAGEM(
    CODIGO_RESERVA NVARCHAR(200) NOT NULL,
    IDA_VOLTA BIT NOT NULL,
    DATA_COMPRA DATE NOT NULL,
    VALOR_TOTAL FLOAT NOT NULL,
    PASSAGEM_IDA BIT NOT NULL,
    PASSAGEM_VOLTA BIT NOT NULL,
    CPF_CLIENTE BIGINT NOT NULL,
    CONSTRAINT PK_CODIGO_RESERVA PRIMARY KEY (CODIGO_RESERVA), 
    CONSTRAINT FK_CPF_CLIENTE FOREIGN KEY (CPF_CLIENTE) REFERENCES CLIENTE (CPF) 
)
